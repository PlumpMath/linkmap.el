#+BEGIN_SRC fundamental
  ;;; LEFT_TABLE_FILE_NAME

  (RECORD RECORD ...)
#+END_SRC

- LEFT_TABLE_FILE_NAME :: filename
- RECORD :: `list' data
- (car RECORD) :: boolean which means whether link is relative or absolute. t is relative.
- (cdr RECORD) :: link map
  - (cadr RECORD) :: source
  - (cddr RECORD) :: target
